# 随机链表的复制

## 题目

## 解题思路
千万注意，慎用 if not，不止是 None 会命中， 0 也会命中！空字符串也会命中！

最优解：原地交错迭代

次优解：哈希 + 递归，哈希的 key 和 value 分别为原节点和新节点

## 代码
**最优解：原地交错迭代**
class Solution(object):
    def copyRandomList(self, head):
        """
        :type head: Node
        :rtype: Node
        """
        if not head:
            return None
        cur = head
        while cur:
            copy = Node(cur.val)
            copy.next = cur.next
            cur.next = copy
            cur = copy.next
        cur = head
        while cur:
            if not cur.random:
                copy.random = None
            else:
                cur.next.random = cur.random.next
            cur = cur.next.next
        copy_head = head.next
        cur = head
        while cur:
            copy = cur.next
            cur.next = copy.next
            if copy.next:
                copy.next = copy.next.next
            cur = cur.next

        return copy_head


**次优解：哈希 + 递归**
class Solution(object):
    def copyRandomList(self, head):
        """
        :type head: Node
        :rtype: Node
        """
        if not head:
            return None
        node_to_copy_dict = {}

        def dfs(node):
            if not node:
                return None
            if node in node_to_copy_dict:
                return node_to_copy_dict[node]
            copy = Node(node.val)
            node_to_copy_dict[node] = copy            
            
            copy.next = dfs(node.next)
            copy.random = dfs(node.random)

            return node_to_copy_dict[node]
        
        return dfs(head)
