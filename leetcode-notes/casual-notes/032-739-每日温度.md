class Solution(object):
    def dailyTemperatures(self, temperatures):
        """
        :type temperatures: List[int]
        :rtype: List[int]
        """
        stack = [0] * len(temperatures)
        index = -1
        res = [0] * len(temperatures)
        for i in range(len(temperatures)):
            tempera = temperatures[i]
            if index < 0 or stack[index][0] >= tempera:
                index += 1
                stack[index] = (tempera, i)
            else:
                while index >= 0 and stack[index][0] < tempera:
                    res[stack[index][1]] = i - stack[index][1]
                    index -= 1
                index += 1
                stack[index] = (tempera, i)

        while (index >= 0):
            res[stack[index][1]] = 0
            index -= 1
        return res
